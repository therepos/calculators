name: Build App Manifest on: push: paths: - 'apps/**.html' - '.github/workflows/manifest.yml' jobs: build-manifest: runs-on: ubuntu-latest permissions: contents: write steps: - uses: actions/checkout@v4 - name: Generate manifest.json run: | echo "[" > apps/manifest.json first=true for f in apps/*.html; do [ -f "$f" ] || continue file=$(basename "$f") slug="${file%.html}" title=$(grep -oP '(?<=).*?(?=)' "$f" | head -1) title="${title:-$slug}" desc=$(grep -oP '> apps/manifest.json; fi printf ' {"slug":"%s","file":"%s","href":"apps/%s","title":"%s"' "$slug" "$file" "$file" "$title" >> apps/manifest.json if [ -n "$desc" ]; then printf ',"description":"%s"' "$desc" >> apps/manifest.json; else printf ',"description":null' >> apps/manifest.json; fi if [ -n "$accent" ]; then printf ',"accent":"%s"' "$accent" >> apps/manifest.json; else printf ',"accent":null' >> apps/manifest.json; fi printf '}' >> apps/manifest.json done echo "" >> apps/manifest.json echo "]" >> apps/manifest.json cat apps/manifest.json - name: Commit manifest run: | git config user.name "github-actions[bot]" git config user.email "github-actions[bot]@users.noreply.github.com" git add apps/manifest.json git diff --cached --quiet && echo "No changes" || (git commit -m "chore: update app manifest" && git push)
